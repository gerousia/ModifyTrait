
namespace = mod_lifestyle_trait_track

scripted_effect mlt_add_defined_tiered_trait_xp_effect = {
	$TARGET$ = {
		if = {
			limit = { NOT = { has_trait = $TRAIT$ } }
			add_trait = $TRAIT$
		}
		if = { 
			limit = {
				has_trait_xp = {
					trait = $TRAIT$
					track = $TRACK$
					value < 100
				}
			}
			add_trait_xp = {
				TRAIT = $TRAIT$
				TRACK = $TRACK$
				value = 100
			}
		}
	}
}

scripted_effect mlt_remove_defined_tiered_trait_xp_effect = {
	$TARGET$ = {
		if = {
			limit = { has_trait = $TRAIT$ }
			remove_trait = $TRAIT$
		}
		if = { 
			limit = {
				NOT = { 
					has_trait_xp = {
						trait = $TRAIT$
						track = $TRACK$
						value = 0
					}
				}
			}
			add_trait_xp = {
				TRAIT = $TRAIT$
				TRACK = $TRACK$
				value = -100
			}
		}
	}
}
